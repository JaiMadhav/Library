CREATE DATABASE db;
USE db;

CREATE TABLE Books (
BId varchar(10) PRIMARY KEY,
Title varchar(50) NOT NULL UNIQUE,
Author varchar(50) NOT NULL,
Available varchar(5) DEFAULT‘YES’);
 
CREATE TABLE Issue (
BId varchar(10) PRIMARY KEY,
Student Name varchar(50) NOT NULL,
FOREIGN KEY(BId) REFERENCES Books(BId) );
 
DESCRIBE Books;
DESCRIBE Issue;
